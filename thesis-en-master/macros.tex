%%% Minor tweaks of style

% These macros employ a little dirty trick to convince LaTeX to typeset
% chapter headings sanely, without lots of empty space above them.
% Feel free to ignore.
\makeatletter
\def\@makechapterhead#1{
  {\parindent \z@ \raggedright \normalfont
   \Huge\bfseries \thechapter\quad #1
   \par\nobreak
   \vskip 20\p@
}}
\def\@makeschapterhead#1{
  {\parindent \z@ \raggedright \normalfont
   \Huge\bfseries #1
   \par\nobreak
   \vskip 20\p@
}}
\makeatother

% This macro defines a chapter, which is not numbered, but is included
% in the table of contents.
\def\chapwithtoc#1{
\chapter*{#1}
\addcontentsline{toc}{chapter}{#1}
}

%%% An environment for typesetting of program code and input/output
%%% of programs.
\DefineVerbatimEnvironment{code}{Verbatim}{fontsize=\small, frame=single}

%%% Shrink a table to \textwidth only if it would otherwise overflow;
%%% narrower tables are left at their natural size instead of being
%%% force-stretched (which \resizebox{\textwidth}{!}{...} alone would do,
%%% blowing up the font of an already-narrow table).
\newsavebox{\shrinkbox}
\newcommand{\shrinktofit}[1]{%
  \sbox{\shrinkbox}{#1}%
  \ifdim\wd\shrinkbox>\textwidth
    \resizebox{\textwidth}{!}{\usebox{\shrinkbox}}%
  \else
    \usebox{\shrinkbox}%
  \fi
}

% Allow long \texttt{LIKE_THIS_IDENTIFIER} terms to break after an
% underscore instead of overflowing into the margin (does not affect
% verbatim text or math-mode subscripts).
\renewcommand{\_}{\textunderscore\penalty0\hskip0pt\relax}
